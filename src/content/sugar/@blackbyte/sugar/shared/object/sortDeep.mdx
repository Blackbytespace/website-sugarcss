## Example
<s-code language="js" header>
```js
import { sortDeep } from '@blackbyte/sugar/object';
sortDeep({
   lolo: { weight: 2 },
   coco: { weight: 10 },
   plop: { weight: 5 },
   aha: {
     hello: 'world',
     coco: 'plop'
   }
}, (a, b) => {
   return a.key.localeCompare(b.key);
});
// {
//   aha: {
//      coco: 'plop',
//      hello: 'world'
//   }
//   coco: { weight: 10 }
//   lolo: { weight: 2 },
//   plop: { weight: 5 },
// }
```

## Example
<s-code language="js" header>
```js
import { __sort } from '@blackbyte/sugar/object';
__sort({
   coco: { weight: 10 },
   lolo: { weight: 2 },
   plop: { weight: 5 }
}, (a, b) => {
  return a.value.weight - b.value.weight;
});
// {
//   lolo: { weight: 2 },
//   plop: { weight: 5 },
//   coco: { weight: 10 }
// }
```

## Example
<s-code language="js" header>
```js
import { diff } from '@blackbyte/sugar/object';
const myObject1 = {
   hello: 'world',
   plop: 'yop'
};
const myObject2 = {
   coco: 'plop',
   hello: 'hey!',
   plop: 'yop'
};
diff(myObject1, myObject2);
{
   coco: 'plop',
   hello: 'hey!'
}
```
